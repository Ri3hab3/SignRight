-- ===================================================================== -- SignRight Database Schema -- Run this entire file in Supabase SQL Editor (one-time setup) -- ===================================================================== -- ===================================================================== -- TABLE 1: profiles -- Extends Supabase's built-in auth.users with app-specific data. -- Auto-created when someone signs up (via the trigger below). -- ===================================================================== create table public.profiles ( id uuid references auth.users(id) on delete cascade primary key, email text unique not null, full_name text, plan text not null default 'free' check (plan in ('free', 'unlimited')), plan_expires_at timestamptz, -- null for free, future date for unlimited subscribers is_admin boolean default false, created_at timestamptz default now() not null, updated_at timestamptz default now() not null ); -- Index for fast lookups by email (login, admin search) create index profiles_email_idx on public.profiles(email); -- ===================================================================== -- TABLE 2: form_purchases -- Records every one-time form purchase ($4.99 NDA, etc.) -- ===================================================================== create table public.form_purchases ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users(id) on delete cascade not null, form_slug text not null, -- e.g. 'nda', 'rental-agreement', 'bill-of-sale' form_name text not null, -- human-readable: 'Non-Disclosure Agreement' amount_usd numeric(10,2) not null, -- e.g. 4.99 payment_provider text not null default 'paypal' check (payment_provider in ('paypal', 'stripe', 'manual')), payment_id text, -- provider's transaction ID for reconciliation status text not null default 'completed' check (status in ('pending', 'completed', 'refunded', 'failed')), created_at timestamptz default now() not null ); create index form_purchases_user_id_idx on public.form_purchases(user_id); create index form_purchases_form_slug_idx on public.form_purchases(form_slug); create index form_purchases_status_idx on public.form_purchases(status); -- ===================================================================== -- TABLE 3: subscriptions -- For the optional "$19.99/month unlimited" plan upsell -- ===================================================================== create table public.subscriptions ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users(id) on delete cascade not null, plan text not null check (plan in ('unlimited_monthly', 'unlimited_annual')), amount_usd numeric(10,2) not null, status text not null default 'active' check (status in ('active', 'cancelled', 'expired', 'past_due')), payment_provider text not null default 'paypal', payment_id text, current_period_start timestamptz not null, current_period_end timestamptz not null, cancelled_at timestamptz, created_at timestamptz default now() not null ); create index subscriptions_user_id_idx on public.subscriptions(user_id); create index subscriptions_status_idx on public.subscriptions(status); -- ===================================================================== -- TABLE 4: documents -- Users' saved/signed documents (the "vault") -- We store metadata here; actual files will live in Supabase Storage (later) -- ===================================================================== create table public.documents ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users(id) on delete cascade not null, name text not null, storage_path text, -- path in Supabase Storage bucket (added later) source text not null default 'upload' check (source in ('upload', 'form', 'template')), form_slug text, -- if created from a form, which one size_bytes bigint, signed boolean default false, created_at timestamptz default now() not null ); create index documents_user_id_idx on public.documents(user_id); create index documents_created_at_idx on public.documents(created_at desc); -- ===================================================================== -- TABLE 5: events -- Server-side event log (backend counterpart to Clarity's frontend events) -- Useful for admin analytics, conversion funnel analysis -- ===================================================================== create table public.events ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users(id) on delete set null, -- null for anonymous events event_name text not null, -- 'signup', 'purchase', 'form_completed', etc. metadata jsonb, -- flexible extra data ip_address inet, user_agent text, created_at timestamptz default now() not null ); create index events_event_name_idx on public.events(event_name); create index events_user_id_idx on public.events(user_id); create index events_created_at_idx on public.events(created_at desc); -- ===================================================================== -- TRIGGER: auto-create profile row when a user signs up -- ===================================================================== create or replace function public.handle_new_user() returns trigger language plpgsql security definer set search_path = public as $$ begin insert into public.profiles (id, email, full_name) values ( new.id, new.email, coalesce(new.raw_user_meta_data->>'full_name', split_part(new.email, '@', 1)) ); return new; end; $$; create trigger on_auth_user_created after insert on auth.users for each row execute function public.handle_new_user(); -- ===================================================================== -- TRIGGER: auto-update "updated_at" on profile changes -- ===================================================================== create or replace function public.set_updated_at() returns trigger language plpgsql as $$ begin new.updated_at = now(); return new; end; $$; create trigger profiles_set_updated_at before update on public.profiles for each row execute function public.set_updated_at(); -- ===================================================================== -- ROW LEVEL SECURITY (RLS) -- Locks down the database so users can only see/modify their own data -- ===================================================================== -- Enable RLS on all tables alter table public.profiles enable row level security; alter table public.form_purchases enable row level security; alter table public.subscriptions enable row level security; alter table public.documents enable row level security; alter table public.events enable row level security; -- -------- profiles policies -------- -- Users can read their own profile create policy "Users can view own profile" on public.profiles for select using (auth.uid() = id); -- Users can update their own profile (but NOT plan, is_admin, or plan_expires_at) create policy "Users can update own profile" on public.profiles for update using (auth.uid() = id) with check (auth.uid() = id); -- Admins can view all profiles create policy "Admins can view all profiles" on public.profiles for select using ( exists ( select 1 from public.profiles where id = auth.uid() and is_admin = true ) ); -- -------- form_purchases policies -------- create policy "Users can view own purchases" on public.form_purchases for select using (auth.uid() = user_id); create policy "Admins can view all purchases" on public.form_purchases for select using ( exists ( select 1 from public.profiles where id = auth.uid() and is_admin = true ) ); -- Note: INSERT into form_purchases happens server-side via webhook with secret key, -- so no public INSERT policy. This prevents users from faking purchases. -- -------- subscriptions policies -------- create policy "Users can view own subscriptions" on public.subscriptions for select using (auth.uid() = user_id); create policy "Admins can view all subscriptions" on public.subscriptions for select using ( exists ( select 1 from public.profiles where id = auth.uid() and is_admin = true ) ); -- -------- documents policies -------- create policy "Users can view own documents" on public.documents for select using (auth.uid() = user_id); create policy "Users can insert own documents" on public.documents for insert with check (auth.uid() = user_id); create policy "Users can update own documents" on public.documents for update using (auth.uid() = user_id); create policy "Users can delete own documents" on public.documents for delete using (auth.uid() = user_id); -- -------- events policies -------- -- Anyone authenticated can log their own events create policy "Users can insert own events" on public.events for insert with check (auth.uid() = user_id or user_id is null); -- Only admins can read events create policy "Admins can view all events" on public.events for select using ( exists ( select 1 from public.profiles where id = auth.uid() and is_admin = true ) ); -- ===================================================================== -- USEFUL VIEWS FOR ADMIN -- These make your admin panel queries trivial -- ===================================================================== -- View: paid users with lifetime spend create view public.admin_user_overview as select p.id, p.email, p.full_name, p.plan, p.plan_expires_at, p.created_at as signed_up_at, coalesce(sum(fp.amount_usd) filter (where fp.status = 'completed'), 0) as lifetime_spend_usd, count(fp.id) filter (where fp.status = 'completed') as total_purchases, max(fp.created_at) as last_purchase_at from public.profiles p left join public.form_purchases fp on fp.user_id = p.id group by p.id, p.email, p.full_name, p.plan, p.plan_expires_at, p.created_at;
